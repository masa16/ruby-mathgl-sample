PROCS['data1'] = proc do
  a = MathGL::MglData.new(40,50,60)
  b = MathGL::MglData.new
  fill(a,"exp(-x^2-4*y^2-16*z^2)")
  light(true)
  alpha(true)
  b.set(a)
  b.diff("x")
  sub_plot(5,3,0)
  b.norm(-1,1,true)
  rotate(70,60)
  box()
  surf3(b)
  b.set(a)
  b.diff2("x")
  sub_plot(5,3,1)
  b.norm(-1,1,true)
  rotate(70,60)
  box()

  b.set(a)
