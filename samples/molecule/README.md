PROCS['molecule'] = proc do
  vertex_color(false)
  compression(false)
  double_sided(false)
  alpha(true)
  light(true)
  sub_plot(2,2,0,"")
  title("Methane, CH_4")
  start_group("Methane")
  rotate(60,120)
  sphere([0,0,0],0.25,"k")
  drop([0,0,0],[0,0,1],0.35,"h",1,2)
  sphere([0,0,0.7],0.25,"g")
  drop([0,0,0],[-0.94,0,-0.33],0.35,"h",1,2)
  sphere([-0.66,0,-0.23],0.25,"g")
  drop([0,0,0],[0.47,0.82,-0.33],0.35,"h",1,2)
  sphere([0.33,0.57,-0.23],0.25,"g")
  drop([0,0,0],[0.47,-0.82,-0.33],0.35,"h",1,2)
  sphere([0.33,-0.57,-0.23],0.25,"g")
  end_group()
  sub_plot(2,2,1,"")
  title("Water, H_{2}O")
  start_group("Water")
  rotate(60,100)
  start_group("Water_O")
  sphere([0,0,0],0.25,"r")
  end_group()
  start_group("Water_Bond_1")
  drop([0,0,0],[0.3,0.5,0],0.3,"m",1,2)
  end_group()
  start_group("Water_H_1")
  sphere([0.3,0.5,0],0.25,"g")
  end_group()
  start_group("Water_Bond_2")
  drop([0,0,0],[0.3,-0.5,0],0.3,"m",1,2)
  end_group()
  start_group("Water_H_2")
  sphere([0.3,-0.5,0],0.25,"g")
  end_group()
  end_group()
  sub_plot(2,2,2,"")
  title("Oxygen, O_2")
  start_group("Oxygen")
  rotate(60,120)
  drop([0,0.5,0],[0,-0.3,0],0.3,"m",1,2)
  sphere([0,0.5,0],0.25,"r")
  drop([0,-0.5,0],[0,0.3,0],0.3,"m",1,2)
  sphere([0,-0.5,0],0.25,"r")
  end_group()
  sub_plot(2,2,3,"")
  title("Ammonia, NH_3")
  start_group("Ammonia")
  rotate(60,120)
  sphere([0,0,0],0.25,"b")
  drop([0,0,0],[0.33,0.57,0],0.32,"n",1,2)
  sphere([0.33,0.57,0],0.25,"g")
  drop([0,0,0],[0.33,-0.57,0],0.32,"n",1,2)
  sphere([0.33,-0.57,0],0.25,"g")
  drop([0,0,0],[-0.65,0,0],0.32,"n",1,2)
  sphere([-0.65,0,0],0.25,"g")
  end_group()
  double_sided( true )
end
